1. Build and push the image:

   docker build -t {{ .Values.image.repository }}:{{ .Values.image.tag }} .
   docker push {{ .Values.image.repository }}:{{ .Values.image.tag }}

2. Install the chart (SECRET_KEY_BASE is required):

   helm upgrade --install {{ .Release.Name }} ./helm-chart \
     --set secretKeyBase=$(openssl rand -hex 64)

3. Access the API:

{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}/api/articles
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "ruby-test-apps.fullname" . }})
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT/api/articles
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "ruby-test-apps.fullname" . }}
{{- else if contains "ClusterIP" .Values.service.type }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "ruby-test-apps.fullname" . }} 3000:{{ .Values.service.port }}
  echo http://127.0.0.1:3000/api/articles
{{- end }}
